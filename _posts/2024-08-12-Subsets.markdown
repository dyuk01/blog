```python
class Solution(object):
    def subsets(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []

        def dfs(nums, path, res):
            res.append(path)
            for i in range(len(nums)):
                dfs(nums[i + 1:], path + [nums[i]], res)
            
        dfs(nums, [], res)
        return res
```

## Time Complexity
O(2<sup>n</sup>)
